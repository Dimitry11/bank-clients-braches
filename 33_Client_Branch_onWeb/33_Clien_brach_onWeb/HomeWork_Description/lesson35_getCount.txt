public static int GetCount(string search)
        {
            int count = -1;
            try
            {
                string command = "SELECT COUNT(*) AS C" +
                                 " FROM[bank].[dbo].[Client]";
                if (!String.IsNullOrEmpty(search))
                {
                    command += " WHERE [Name] LIKE N'%" + search + "%' OR Id_client LIKE '%" + search + "%'";
                }

                using (SqlConnection conn = new SqlConnection(Config.connectionString))
                {
                    using (SqlCommand cmd = new SqlCommand(command, conn))
                    {
                        conn.Open();
                        SqlDataReader sqlReader = cmd.ExecuteReader();
                        while (sqlReader.Read())
                        {
                            string _count = sqlReader["C"].ToString();
                            count = int.Parse(_count);
                        }
                        sqlReader.Close();
                        conn.Close();
                    }
                }
            }
            catch (Exception ex)
            {
                throw ex;
            }
            return count;
        }
